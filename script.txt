CREATE TABLE USUARIOS(
IDUSUARIO int primary key AUTO_INCREMENT,
CI VARCHAR(20) NOT NULL,
NOMBRES VARCHAR(60) NOT NULL,
APELLIDOS VARCHAR(60) NOT NULL,
TELEFONO VARCHAR(20),
CORREO VARCHAR(60),
CONTRASENA VARCHAR(20),
ROL CHAR(1),
ESTADO VARCHAR(10) DEFAULT 'NORMAL',
EJEX VARCHAR(50) NOT NULL,
EJEY VARCHAR(50) NULL,
ACTIVO CHAR(1) default '1'
)

DONDE ROL
1: USUARIO NORMAL
2: ADMINISTRADOR

DONDE ESTADO
NORMAL: PERSONA SIN CORONAVIRUS
ACTIVO: PORTADOR DE CORONAVIRUS
RECUPERADO: PERSONA QUE SE CURO DEL CORONAVIRUS
FALLECIDO: PERSONA QUE MURIO CON CORONAVIRUS
